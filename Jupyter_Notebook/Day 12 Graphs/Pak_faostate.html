import seaborn as sns
import pandas as pd
sns.set_theme(style="whitegrid")

fertilizer= pd.read_csv("fertilizer.csv")

# Draw a nested barplot by Area_code and values
g = sns.catplot(
    data=fertilizer, kind="bar",
    x="Area Code", y="Value", hue="Item",
    ci="sd", palette="dark", alpha=.6, height=6
)
g.despine(left=True)
g.set_axis_labels("", "Area")
g.legend.set_title("")
